usage:
    ctc record help SUBCOMMAND [options]

description:
    record help output to an html or svg file

arguments:
    SUBCOMMAND          subcommand to record help of
    --path PATH         output path to html or svg file
    --overwrite         whether to overwrite files if they already exist
    --all               record help for all subcommands
    --include-hidden    if using --all, include hidden subcommands
