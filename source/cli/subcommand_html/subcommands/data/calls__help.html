usage:
    ctc calls ARGS [options]

description:
    output the result of multiple calls

examples:
    # version 1: same call across multiple blocks
    ctc calls <address> <function_name> [<function_parameters>] --blocks 
<blocks>

    # version 2: same call across multiple addresses
    ctc calls <function_name> [<function_parameters>] --addresses <addresses> 
[--block block]

arguments:
    ARGS                     <see above>
    --addresses ADDRESSES    addresses to point calls toward
    --blocks BLOCKS          block range for calls
    --block BLOCK            block number for calls
    --quiet                  omit summary, only output function result
    --output OUTPUT          file path for output (.json or .csv)
    --overwrite              specify that output path can be overwritten
    --from FROM              address that calls should come from
