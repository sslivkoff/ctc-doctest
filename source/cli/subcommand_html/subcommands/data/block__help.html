usage:
    ctc block [BLOCK] [options]

description:
    summarize block

examples:
    ctc block 14000000
    ctc block 14000000 --json
    ctc block --timestamp 1600000000

arguments:
    BLOCK                    block number
    --timestamp TIMESTAMP    specify block by timestamp
    --verbose                emit extra information
    --json                   output block as raw json
